# NGINX Font Optimization Configuration
# Add this to your nginx.conf or server block

# Font files caching and headers
location ~* \.(woff2|woff|ttf|otf|eot)$ {
    # Cache for 1 year
    expires 1y;
    
    # Set cache control headers
    add_header Cache-Control "public, immutable, no-transform";
    
    # Enable CORS for fonts
    add_header Access-Control-Allow-Origin "*";
    
    # Enable gzip compression
    gzip on;
    gzip_types font/woff2 font/woff font/ttf font/otf application/vnd.ms-fontobject;
    gzip_comp_level 6;
    
    # Disable access log for fonts (optional, for performance)
    access_log off;
}
